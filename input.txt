// This program uses for loops and block scopes<|SPACE|>// to print the updates to the same variable
var foo = "after";
{
  var foo = "before";

  for (var foo = 0; foo < 1; foo = foo + 1) {
    print foo;
    var foo = -1;
    print foo;
  }
}

{
  for (var foo = 0; foo > 0; foo = foo + 1) {}

  var foo = "after";
  print foo;

  for (foo = 0; foo < 1; foo = foo + 1) {
    print foo;
  }
}
